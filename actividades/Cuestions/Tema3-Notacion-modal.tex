(indica a resposta correcta)
    \begin{enumerate}[a)]
    
    \item Situar os neumas nunha pauta de catro liñas paralelas que marbana a distancia dunha terceira

    \item Situar os neumas nunha pauta de cinco liñas paralelas que marcaban unha distancia de segunda
    
    \item Situar os neumas nunha pauta de cinco liñas paralelas que marcaban unha distancia de terceira
    
    \end{enumerate}
